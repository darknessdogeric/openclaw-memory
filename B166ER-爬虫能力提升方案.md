## 五、工具整合脚本

### 5.1 统一爬虫接口

```python
# b166er_crawler.py - 统一爬虫接口

import requests
from newspaper import Article

class B166ERCrawler:
    """B166ER统一爬虫接口"""
    
    def __init__(self):
        self.tools = {
            'jina': self._jina_read,
            'newspaper': self._newspaper_read,
            'tavily': self._tavily_search,
            'duckduckgo': self._ddg_search,
        }
    
    def read_url(self, url, method='auto'):
        """读取URL内容"""
        if method == 'auto':
            # 智能选择
            return self._smart_read(url)
        
        tool = self.tools.get(method)
        if tool:
            return tool(url)
        
        return None, f"Unknown method: {method}"
    
    def _jina_read(self, url):
        """使用Jina AI读取"""
        try:
            response = requests.get(f"https://r.jina.ai/http://{url}", timeout=30)
            return response.text, "success"
        except Exception as e:
            return None, str(e)
    
    def _newspaper_read(self, url):
        """使用Newspaper3k读取"""
        try:
            article = Article(url)
            article.download()
            article.parse()
            return {
                'title': article.title,
                'text': article.text,
                'date': article.publish_date,
                'authors': article.authors
            }, "success"
        except Exception as e:
            return None, str(e)
    
    def _smart_read(self, url):
        """智能选择最佳工具"""
        # 尝试顺序: Jina -> Newspaper -> 其他
        for method in ['jina', 'newspaper']:
            result, status = self.tools[method](url)
            if status == "success":
                return result, method
        
        return None, "All methods failed"
    
    def search(self, query, engine='auto'):
        """搜索查询"""
        if engine == 'auto':
            # 先用Tavily，失败用DuckDuckGo
            result, status = self._tavily_search(query)
            if status == "success":
                return result, "tavily"
            
            result, status = self._ddg_search(query)
            if status == "success":
                return result, "duckduckgo"
            
            return None, "All search engines failed"
    
    def _tavily_search(self, query):
        """使用Tavily搜索"""
        # 使用现有API Key
        try:
            from tavily import TavilyClient
            client = TavilyClient(api_key="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")
            result = client.search(query)
            return result, "success"
        except Exception as e:
            return None, str(e)
    
    def _ddg_search(self, query):
        """使用DuckDuckGo搜索"""
        try:
            from duckduckgo_search import DDGS
            with DDGS() as ddgs:
                results = ddgs.text(query, max_results=10)
                return list(results), "success"
        except Exception as e:
            return None, str(e)

# 使用示例
crawler = B166ERCrawler()
content, method = crawler.read_url("https://example.com/article")
results, engine = crawler.search("Marriott service standards")
```
